	function reSearch() {
		var qword = $("#q").val();
		if(qword!=""&&bannedHashSet.contains(qword)){
			alert("您的检索词中包含不当词汇，请更换或删除敏感词");
			return;

		if(q == "") {
			alert("请输入检索词.");

	function loginTo(toPage){
		$("#toPageUrl").val(toPage);
		var res =false;
		$.ajax({
			type:'get',
			dataType:'json',
			async : false,
			url:'/opac3/reader/getReaderInfo?return_fmt=json',
			success:function(data){
				if(data.reader==null||data.reader=='undefined'){
					var reader = "";
					var rdLoginId = "";
					var rdEmail = "";
					if(reader != null && (rdLoginId == null || rdLoginId == '' || rdEmail == null || rdEmail == '')){

						$(".perIconArea").trigger("click");
					}
				}else{
					if(toPage!=""){
						window.location.href = toPage;
					}
					res = true;
				}
			},
			error: function(){
				
				
				
				$(".perIconArea").trigger("click");
				
				
			}
		});
		return res;
	}

function sendEmail($clickButton){
	var email = $clickButton.parent().siblings("input[name='email']").first().val();
	if(email=="" || email.length==0) {
		alert("请输入邮箱地址");
		return ;
	}
	var reg = /^([a-z0-9_\.-]+)@([\da-z\.-]+)\.([a-z\.]{2,6})$/;
	if(!reg.test(email)) {
		alert("请输入正确的邮箱地址");
		return ;
	}

	var bookList = $clickButton.parent().siblings("input[name='bookIdList']").first().val();
	var subject = $clickButton.parent().siblings("input[name='subject']").first().val();
	var postdata = {
		"displayType" : "full",
		"destination" : "email",
		"email" : email,
		"subject" : subject,
		"bookIdList" : bookList
	};
	var postUrl = "/opac3/api/book/onlineBookListSendToEmail";
	//先禁止一下提交按钮
	$clickButton.attr("disabled", "disabled");
	$.ajax({
		url : postUrl,
		type : "POST",
		data : postdata,
		dataType : "text",
		error : function(x,s,e) {
			alert("操作失败，请重试。");
		}
	}).done(function(result){
		$clickButton.removeAttr("disabled");
		if(result == "1") {
			alert("发送邮件成功，请注意查看您的邮箱!");
			$clickButton.parent().siblings("input[name='email']").first().val("");
			$(".emailArea").hide();	//发送成功后隐藏
		} else if(result == "0"){
			alert("发送邮件失败，请检查您的邮件设置!");
		} else {
			var ret = eval(result);
			alert(ret);
		}
	});
}
//添加收藏
function addPrivateCollection(bookrecno){
    var checking = checkLogin();
    if(checking){
		//通过样式判断是否选删除
		var isDeleteAction = false;
		var url = "/opac3/privateCollection/add/"+bookrecno+"";
		if ($("#likeid_"+bookrecno).hasClass("thHasLikeIcon")) {
			url = "/opac3/privateCollection/delete?bookrecnoList="+bookrecno+"";	//取消收藏url
			isDeleteAction = true;
		}
		//先实现样式改善体验
		if (isDeleteAction == false) {
			$("#likeid_"+bookrecno).attr("title","取消收藏");
			$("#likeid_"+bookrecno).removeClass("thLikeIcon").addClass("thHasLikeIcon");
		}else{
			$("#likeid_"+bookrecno).attr("title","收藏");
			$("#likeid_"+bookrecno).removeClass("thHasLikeIcon").addClass("thLikeIcon");
		}

        if (isDeleteAction == true) {	//删除动作的，给个提示
            var isEnable = confirm("确定取消收藏吗？");
            if (!isEnable) {
                return;
            }
        }

        //作个标识，防止多次点击样式变乱
        var dataSign = $("#likeid_"+bookrecno).attr("data_sign");
        if(dataSign == 1) {
            alert("操作过于频繁，请稍后尝试！")
            return;
        }
        $("#likeid_"+bookrecno).attr("data_sign",1);

        $.ajax({
            type:"post",
            url:url,
            success:function(result){
                $("#likeid_"+bookrecno).attr("data_sign",0);		//修改标识
            },
            error:function(){
                $("#likeid_"+bookrecno).attr("data_sign",0);		//修改标识
                if (isDeleteAction == true) {
                    $("#likeid_"+bookrecno).attr("title","取消收藏");
                    $("#likeid_"+bookrecno).parent().addClass("hasLiked");
                }else{
                    $("#likeid_"+bookrecno).attr("title","收藏");
                    $("#likeid_"+bookrecno).parent().removeClass("hasLiked");
                }
                alert("后台发生错误!");
            }
        });
    }
}

function dealResercationHuaWei(url) {
	$.ajax({
		type: "get",
		url: url,
		dataType: "json",
		success: function(data) {
			if (data) {
				if (isEmpty(data.message)) {
					var barcode = data.holdingList[0].barcode;
					var pickupcode = data.pickupLocalMap[0].localcode;
					$.ajax({
						type: "get",
						url: "/opac3/reservation/doReservation?return_fmt=json",
						data: {
							barcode: barcode,
							pickupLocal: pickupcode,
							furl: "/opac3/reservation/currentReservationList"
						},
						dataType: "json",
						success: function(data) {
							if (data) {
								if (data.isReservationSuc) {
									$(".appSucPopArea").show();
								} else {
									alert(data.message + " " + data.messages);
								}
							} else {
								alert("服务端异常");
							}
						}
					});
				} else {
					alert(data.message)
				}
			} else {
				alert("服务端异常")
			}
		}
	});
}

function getCurlibcodes(){
	var curLibcodes = "";
	var navPageBaseURL = "searchType=standard&isFacet=true&view=standard&rows=10&sortWay=score&sortOrder=desc&hasholding=1&f_curlibcode=BEILIN&searchWay0=title&searchWay1=author&searchWay2=marc&q0=%E5%A6%82%E4%BD%95%E9%98%85%E8%AF%BB%E4%B8%80%E6%9C%AC%E4%B9%A6&logical0=AND&logical1=AND&logical2=AND";
	var index = navPageBaseURL.indexOf("f_curlibcode");
	var lastindex = -1;
	while(index != -1){
		navPageBaseURL = navPageBaseURL.substring(index+13);
		lastindex = navPageBaseURL.indexOf("&");
		if(lastindex == -1){
			curLibcodes = navPageBaseURL + "," + curLibcodes;
		}else{
			curLibcodes = navPageBaseURL.substring(0,lastindex) + "," + curLibcodes;
		}
		index = navPageBaseURL.indexOf("f_curlibcode");
	}
	if(curLibcodes.endsWith(",")){
		curLibcodes = curLibcodes.substring(0,curLibcodes.length-1);
	}
	return curLibcodes;
}

function getUrlParameter(url, name){
	var r = null;
	var urls = url.split("&");
	var res = '';
	for (var urlNum in urls) {

		if(urls[urlNum].indexOf('_') == -1){
			var reg = new RegExp(name + "=([^"+name+"]*)");
			r = urls[urlNum].match(reg);
			if (r != null && r != undefined) {
				res += r[1] + ",";
			}
		}
	}
	if(res != null) return res; return null;
};
function createTable(className, tableHeader, tableContent) {
	var url = window.location.href;
	var name = "curlibcode";

	var urlLibcodes = urlLibcode.split(',');

	var s = '<table border="1" cellspacing="0" cellpadding="0" class="'+className+'"><tr>';
	for(index in tableHeader) {
		s += "<th>" + tableHeader[index] + "</th>";
	}
	s += "</tr>";
	for(index in tableContent) {
		s += "<tr>";
		var recordData = tableContent[index];
		var glc="P2SXS029038";
		if (urlLibcode == "" || urlLibcode == null){
			for(j in recordData) {
				if (j > 0) {
					var style = "";
					if (glc == "P3JS0519059" && j == 0) {
						style = " style='color:red;'";
					}
					s += "<td" + style + ">" + recordData[j] + "</td>";
				}
			}
		} else {
			for(j in recordData) {
				if (j == 0) {
					var curlib = recordData[j];
				}
				for (var libIdx=0;libIdx<urlLibcodes.length;libIdx++) {

						}
					}

function showBookLoanAndBookAppoint(bookrecnos) {
    var tranBookrecnos = "";
    for (var i = 0;i <bookrecnos.length;i++ ){
        tranBookrecnos = tranBookrecnos + bookrecnos[i] + ",";
    }
    var url = "/opac3/api/holding/getHoldingsBybookrecnos";
    var requestData = {"bookrecnos":tranBookrecnos};
    $.post(url, requestData, function (data) {
        var isChangeCurLoacl = "0";
        var localMap = {};
        if (isChangeCurLoacl!=1){
            localMap = data.localMap;
        }
        var libcodeDeferDateMap = data.libcodeDeferDateMap;

        var holdingListMap = data.holdingListMap;
        if (holdingListMap) {
			for(var bookrecno in holdingListMap){
				var holdings = holdingListMap[bookrecno];
				var holdMap = new Array();
				jQuery.each(holdings,function(n,holding){
					var map={lib:holding.curlib,local:holding.curlocal,
						holdings:new Array(holding)};
					var flag = true;
					for(var i in holdMap){
						if(holdMap[i].lib==holding.curlib&&holdMap[i].local==holding.curlocal){
							holdMap[i].holdings.push(holding);
							flag = false;
							break;
						}
					}
					if(flag){
						holdMap.push(map);
					}
				});
				for(var i in holdMap){
					if(i=='array_contains') continue;

					var holdings = holdMap[i].holdings;
					var curlib = holdMap[i].lib;

					var outOfLibNum = 0;
					var inLibNum = 0;
					for(var j in holdings){
						var status = holdings[j].state;
						var indate = holdings[j].indate;

						//状态
						var curlibDeferDays = libcodeDeferDateMap[curlib];
						if(status == 2 && curlibDeferDays > 0) {
							var currentDate = new Date();
							var todayDate = new Date(currentDate.getFullYear(),
								currentDate.getMonth(), currentDate.getDate());
							var dateStr = indate.split("-");
							var bookDate = new Date(dateStr[0],dateStr[1],dateStr[2]).getTime();
							var intervalDays = new Number((todayDate-bookDate) / (24*60*60*1000));
							var deferDays = curlibDeferDays + 1;
							if(intervalDays < deferDays) {
								//还没过上架天数，则改变显示状态为在编状态
								status = 1;
							}
						}
						//借出时显示应还时间
						if(status==3){
							outOfLibNum++;
						}else if(status==2){
							inLibNum++;
						}
						if(outOfLibNum > 0){
							//$(".appointBook").show(); //只要存在[借出]的书籍则显示预约按钮
							$("#holdingPreview_" + bookrecno).parents(".rankLiIn").find(".appointBook").show();
						}
						if(inLibNum > 0){
							//$(".bookLoan").show(); 	 //只要存在[在馆]的书籍则显示预借按钮
							$("#holdingPreview_" + bookrecno).parents(".rankLiIn").find(".bookLoan").show();
						}
					}
				}
			}
        }
    });
}

function addPrelendBookList(bookrecno,libcode,isRedirect) {
	if(!canPrelendListByBookrecno(libcode,bookrecno)){
        $("#prelendTab").show();
		$('html,body').removeClass('overHide');
		return false;
	}
	$(".resultTips").html("预借成功!");
	$(".success_tips").html("您可到预借书架进行预借！");
	$('html,body').removeClass('overHide');
	$.ajax({
		url:"/opac3/prelend/bookshelf",
		type: "POST",
		async:false,
		data:{"bookrecno":bookrecno,"libcode":libcode},
		success: function(data){
			if( data.code == '200'){
				$(".tips").show();
				setTimeout("$(\".tips\").hide();",2000);
			}
			if(isRedirect && data.code == '200'){
				document.location.href = '/opac3/prelend/CanPrelendListByBookrecno?prelendbookshelfIds='+data.data;
			}
		}
	});
}

function canPrelendListByBookrecno(libcode,bookrecno){
	var flag = false;
	$.ajax({
		url:"/opac3/prelend/getCanPrelendListByBookrecno/",
		type: "POST",
		async:false,
		data:{"bookrecno":bookrecno,"libcode":libcode},
		success: function(data){
			if(data.length ==0){
				$(".resultTips").html("借阅申请失败");
				$(".success_tips").html("当前所选分馆暂无可预借的馆藏");
			}else{
				var message = new Array();
				for(var i=0;i<data.length;i++){
					if(data[i].canPrelend){
						flag = true;
						break;
					}else{
						message = message.concat(data[i].msgList);
					}
				}
				if(!flag){
					$.unique(message);
					if(message.length>3){message = message.slice(0,3);}
					$(".resultTips").html("借阅申请失败");
					$(".success_tips").html(message.join("&nbsp;"));
				}
			}
		}
	});
	return flag;
}
